⚡ koprobe

Kubernetes Cost Attribution via eBPF — real kernel-level usage, not estimates.
eBPF Kubernetes v0.1.0 Apache 2.0

Add Helm repository

helm repo add koprobe https://mihir99-mk.github.io/koprobe
helm repo update

Install

helm install koprobe koprobe/koprobe \
  --namespace monitoring \
  --create-namespace \
  --set cloud.provider=aws \
  --set cloud.region=us-east-1

Available charts

helm search repo koprobe
# NAME              CHART VERSION   APP VERSION   DESCRIPTION
# koprobe/koprobe   0.1.0           0.1.0         K8s Cost Attribution via eBPF

Configuration

helm install koprobe koprobe/koprobe \
  --set cloud.provider=gcp \        # aws | gcp | azure
  --set cloud.region=us-central1 \
  --set slack.enabled=true \
  --set slack.webhookURL=https://hooks.slack.com/...
